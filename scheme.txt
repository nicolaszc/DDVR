/index.html          ← entrypoint de la SPA

/assets/
└── js/              ← librerías locales (offline: jQuery, Isotope, QR, etc.)

/pages/
├── js/
│   ├── router/
│   │   ├── include.js   ← carga HTML de componentes de manera asíncrona
│   │   ├── router.js    ← controla la navegación SPA (PLP <-> PDP)
│   │   └── main.js      ← entrypoint principal del frontend, inicializa todo
│   ├── app.js           ← funciones globales UI, animaciones, helpers
│   ├── plp.js           ← lógica específica de PLP: render, filtros, isotope
│   └── pdp.js           ← lógica específica de PDP: render producto, random cards
├── plp.html             ← markup de catálogo (PLP)
├── pdp.html             ← markup de ficha producto (PDP)

/components/
├── js/
│   ├── form-post.js     ← validación y envío de formularios
│   └── qr-btns.js       ← generación de QR y acciones de share
└── otros componentes... ← header, footer, card, botones fijos, theme toggle, etc.

/data/
├── automotive.json      ← productos automotriz
└── industrial.json      ← productos industrial

/api/
├── qr/
│   └── qr-get.php       ← genera QR dinámicos de cada producto
└── mail/
    └── form-post.php    ← recibe formularios de contacto

/admin/
├── login.php            ← autenticación admin
├── upload.php           ← subida de archivos (img/pdf)
└── editor.php           ← editor de contenido (Monaco)

/uploads/
├── img/
│   ├── automotriz
│   └── industrial
└── pdf/
    ├── automotriz
    └── industrial
